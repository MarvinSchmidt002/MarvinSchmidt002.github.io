Music dodge is a game where you dodge obstacles to the beat of the music. This project was created for the application process of the University of Applied Sciences Darmstadt.

Obstacles

There is four different obstacles that spawn to the beat of the music. The player can dodge them by moving to the left or right, jumping and ducking.

To the Beat

To make the obstacles spawn to the beat of the music I used the audioSource data and the bpm of the song to calculate intervals for the obstacles to spawn.

public class BeatManager : MonoBehaviour
{
    [SerializeField] private float bpm;
    [SerializeField] private AudioSource audioSource;
    [SerializeField] Intervals[] intervals;
    
    private void FixedUpdate()
    {
        foreach (Intervals interval in intervals)
        {
            //syncs the interval with the music
            float sampledTime = (audioSource.timeSamples /
                (audioSource.clip.frequency *
                 interval.GetIntervalLength(bpm) ) );
            //checks if the interval has changed
            interval.CheckForNewInterval(sampledTime);
        }
    }
}

[System.Serializable]
public class Intervals 
{
    [SerializeField] private float steps;
    [SerializeField] private UnityEvent trigger;
    private int lastInterval;
    
    /*returns the length of the interval in seconds
     based on the bpm and the steps provided.
     Higher steps equal longer intervals.*/
    public float GetIntervalLength(float bpm)
    {
        return 60f / (bpm * steps);
    }
    
    //triggers everytime the interval changes
    public void CheckForNewInterval (float interval)
    {
        if(Mathf.FloorToInt(interval) != lastInterval)
        {
            lastInterval = Mathf.FloorToInt(interval);
            trigger.Invoke();
        }
    }
}
