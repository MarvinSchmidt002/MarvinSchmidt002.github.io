Event System

  • Tags: 
  • Unity

A Scriptable Object based Event System for Unity

Datatypes

First we establish the datatypes that we want to use for our events.

//The data types that are possible to pass to the event
//Update editor script when adding new types
public enum InputType
{
    None,
    Bool,
    Int,
    Float,
    String,
    Vector3,
    Transform,
    GameObject,
}

[Serializable]
public class InputDatatypes
{
    public InputType inputType;
    public bool boolValue;
    public int intValue;
    public float floatValue;
    public string stringValue;
    public Vector3 vector3Value;
    public Transform transformValue;
    public GameObject gameObjectValue;
}

In addition to the input types, we also need a class to store the outputEvents because unity has different kind of built-in unity events, this will make more sense later.

[Serializable]
public class OutputEvents
{
    public InputType inputType;
    public UnityEvent unityEvent;
    public UnityEvent<bool> boolEvent;
    public UnityEvent<int> intEvent;
    public UnityEvent<float> floatEvent;
    public UnityEvent<string> stringEvent;
    public UnityEvent<Vector3> vector3Event;
    public UnityEvent<Transform> transformEvent;
    public UnityEvent<GameObject> gameObjectEvent;
}

Scriptable Object

Next lets take a look at the actual scriptable object that will hold the events.
First we establish some interfaces that we will use for the event and the listener to make sure that we can easily add new events and listeners in the future. The event itself is rather simple, the most important part is that we are setting the input type here that will get synced across all listeners and invokers of the event.

public interface IEvent
{
    void AddListener(IEventListener listener);
    void RemoveListener(IEventListener listener);
    void Invoke();
    void Invoke(object data);
}

public interface IEventListener
{
    void OnInvoke();
    void OnInvoke(object data);
    void Subscribe();
    void Unsubscribe();
}

[CreateAssetMenu(fileName = "ScriptableDataEvent", menuName = "Scriptables/ScriptableDataEvent")]
public class ScriptableEvent : ScriptableObject, IEvent
{
    //Description of the event
    [SerializeField, TextArea(3, 10)]
    private string description;
    
    //Type of data that can be passed to the event
    public InputType inputType;
    
    //List of subscribed Listener
    private List<IEventListener> listeners = new ();

    
    //Getter for the list of listeners
    public IEnumerable<IEventListener> Listeners => listeners;
    
    //Add a listener to the list
    public static ScriptableEvent operator +(ScriptableEvent instance, IEventListener listener)
    {
        instance.AddListener(listener);
        return instance;
    }
    
    //Add a listener to the list
    public void AddListener(IEventListener listener)
    {
        if (listener == null) return;
        listeners.Add(listener);
    }
    
    //Remove a listener from the list
    public static ScriptableEvent operator -(ScriptableEvent instance, IEventListener listener)
    {
        instance.RemoveListener(listener);
        return instance;
    }
    
    //Remove a listener from the list
    public void RemoveListener(IEventListener listener)
    {
        if (listener == null) return;
        listeners.Remove(listener);
    }
    
    //Invoke the event with no data
    public void Invoke()
    {
        foreach (var listener in Listeners)
        {
            listener.OnInvoke();
        }
    }
    //Invoke the event with data
    public void Invoke(object data)
    {
        foreach (var listener in Listeners)
        {
            listener.OnInvoke(data);
        }
    }
}

Event Listener

Next we need a class that will listen to the event and invoke the correct unity event based on the input type. For that we create the EventListenerData class that implements the IEventListener interface which handles the subscription and unsubscription of the event, as well as the invocation of the event with the correct data type. The actual EventListener class is a MonoBehaviour that holds a list of EventListenerData objects and makes sure they’re synced with the input type of the event. It also has a general response event that gets invoked when any event is invoked.

[Serializable]
public class EventListenerData : IEventListener
{
    //Event to listen to
    public ScriptableEvent _event;
    public OutputEvents ResponseEvents;
    public event Action<object> ActionResponse;

    public void Subscribe() => _event += this;
    public void Unsubscribe() => _event -= this;
    
    //invoke the event with no data
    public void OnInvoke()
    {
        ResponseEvents?.unityEvent.Invoke();
        ActionResponse?.Invoke(null);
    }
    //invoke the event with data
    public void OnInvoke(object data)
    {
        switch (ResponseEvents.inputType)
        {
            case InputType.Bool:
                ResponseEvents?.boolEvent.Invoke((bool)data);
                break;
            case InputType.Int:
                ResponseEvents?.intEvent.Invoke((int)data);
                break;
            case InputType.Float:
                ResponseEvents?.floatEvent.Invoke((float)data);
                break;
            case InputType.String:
                ResponseEvents?.stringEvent.Invoke((string)data);
                break;
            case InputType.Vector3:
                ResponseEvents?.vector3Event.Invoke((Vector3)data);
                break;
            case InputType.Transform:
                ResponseEvents?.transformEvent.Invoke((Transform)data);
                break;
            case InputType.GameObject:
                ResponseEvents?.gameObjectEvent.Invoke((GameObject)data);
                break;
        }
        ActionResponse?.Invoke(data);
    }

}

[ExecuteAlways]
public class EventListener : MonoBehaviour, IEventListener
{
    [SerializeField] private List<EventListenerData> _eventsListeners = new ();
    [SerializeField] private UnityEvent GeneralResponse;
    
    public event Action GeneralActionResponse;
    
    //subscribe and unsubscribe depending on the state of the object
    private void OnEnable()
    {
        Subscribe();
    }
    private void OnDisable()
    {
        Unsubscribe();
    }

    //subscribe to all events
    public void Subscribe()
    {
        if (_eventsListeners.Count == 0)
        {
            Debug.LogWarning($"No events defined for {name}! " + "The Event will not be added");
            return;
        }
        foreach (var eventsListener in _eventsListeners)
        {
            eventsListener._event += this;
            eventsListener.Subscribe();
        }
    }
    
    //unsubscribe from all events
    public void Unsubscribe()
    {
        foreach (var eventsListener in _eventsListeners)
        {
            eventsListener._event -= this;
            eventsListener.Unsubscribe();
        }
    }
    
    //invoke the event with no data
    public void OnInvoke()
    {
        GeneralResponse.Invoke();
        GeneralActionResponse?.Invoke();
    }
    public void OnInvoke(object data)
    {
        GeneralResponse.Invoke();
        GeneralActionResponse?.Invoke();
        Debug.Log("Data: " + data);
    }
    
    //TODO slightly unoptimized
    private void Update()
    {
        foreach (var eventsListener in _eventsListeners.Where(eventsListener => eventsListener._event != null))
        {
            eventsListener.ResponseEvents.inputType = eventsListener._event.inputType;
        }
    }
    
}

Event Invoker

Lastly we need a class that will invoke the event with the correct data type. This is just an example of how you could invoke the events with their matching data types.

[ExecuteAlways]
public class EventInvoker : MonoBehaviour
{
    [SerializeField] private List<DataEvents> _dataEvents = new();

    [Serializable]
    public class DataEvents
    {
        public ScriptableEvent Dataevents;
        public InputDatatypes InputDatatypes;
    }
    #if UNITY_EDITOR
    [Button("Invoke Events")]
    #endif
    
    public void InvokeEvents()
    {
        if(_dataEvents.Count == 0)
        {
            Debug.LogWarning("No events to invoke!");
            return;
        }
        //Invoke event with selected data
        foreach (var i in _dataEvents)
        {
            if (i != null)
            {
                switch (i.InputDatatypes.inputType)
                {
                    case InputType.None:
                        i.Dataevents.Invoke();
                        break;
                    case InputType.Bool:
                        i.Dataevents.Invoke(i.InputDatatypes.boolValue);
                        break;
                    case InputType.Int:
                        i.Dataevents.Invoke(i.InputDatatypes.intValue);
                        break;
                    case InputType.Float:
                        i.Dataevents.Invoke(i.InputDatatypes.floatValue);
                        break;
                    case InputType.String:
                        i.Dataevents.Invoke(i.InputDatatypes.stringValue);
                        break;
                    case InputType.Vector3:
                        i.Dataevents.Invoke(i.InputDatatypes.vector3Value);
                        break;
                    case InputType.Transform:
                        i.Dataevents.Invoke(i.InputDatatypes.transformValue);
                        break;
                    case InputType.GameObject:
                        i.Dataevents.Invoke(i.InputDatatypes.gameObjectValue);
                        break;
                }
            }
        }
    }
    //TODO slightly unoptimized
    private void Update()
    {
        foreach (var i in _dataEvents)
        {
            if (i.Dataevents != null)
            {
                i.InputDatatypes.inputType = i.Dataevents.inputType;
            }
        }
    }
}

Example

Here is an example of how you could use the event system in your project.
First we create a new scriptable event, in this case it carries a string value.

Next we create an event listener that listens to the event and invokes a unity event with the string value to change a text component directly without the need of an additional script, since built-in unity events can access the public functions with matching parameter input types.
Lastly we just need to invoke the event with any string value we choose. In this case we are going to use the Event Invoker script from earlier.
That is it, now you can easily create new events and listeners without the need of additional scripts or references to other objects.
