Magic System

The Magic System is based on elemental Ofudas (japanese paper talismans):

The Elements are very modular and are just declared in an enum, so integrating new elements is very easy. 
public static class Elements
{
    //The enum that controls the elemental types available everywhere

//resets the ofuda to its starting state
private void resetOfuda()
{
    hasRolledElement = false;
    hasUsedSpell = false;
    OfudaRenderer.material = Empty;
    //Simple dissolve effect
    StartCoroutine(MaterialDissolver.dissolve(dissolveEnd, dissolveStart, dissolveTime, Empty, "_DissolveManualProgress"));
}

//rolls a random Element of the available ofuda elements
private void rollOfuda()
{
    int random = Random.Range(0, ofudaObjects.Count);
    currentSpellSet = random;
    element = ofudaObjects[random]._Element;
    hasRolledElement = true;
    assignElementMaterial();
}

    ofudaObjects[currentSpellSet]._Material.SetFloat("_DissolveManualProgress", dissolveStart);

}

When you proceed to activate it, it will spawn the spell selection wheel. Spell Selection Wheel: when you activate an Ofuda it records the position of the controller at that moment and opens a four way menu for you to select a spell.

//activates the spell ofuda leading into spell selection
private void spellActivate()
{
    //simple dissolve effect
    StartCoroutine(MaterialDissolver.dissolve(dissolveStart, dissolveEnd, dissolveTime, currentMaterial, "_DissolveManualProgress"));
    spawnSpellSelection();
}

//spawns the spell selection UI
private void spawnSpellSelection()
{
    selector = Instantiate(spellSelection,transform.position,transform.rotation * spellSelection.transform.rotation);
    savedPosition = selector.transform.position;
    selector.transform.SetParent(_player.transform);
    isSelecting = true;
    Directions[0] = selector.transform.GetChild(0).gameObject;
    Directions[1] = selector.transform.GetChild(1).gameObject;
    Directions[2] = selector.transform.GetChild(2).gameObject;
    Directions[3] = selector.transform.GetChild(3).gameObject;
    
    StartCoroutine(SelectionCheck());
    StartCoroutine(materialChanger());
}

//checks for which spell is being selected in the spell selection wheel
IEnumerator SelectionCheck()
{
    float[] array = new float[4];
    
    while (isSelecting)
    {
        //up
        array[0] = Vector3.Distance(Directions[0].transform.position + Directions[0].transform.up * selectionSizeAdjust, transform.position);
        //down
        array[1] = Vector3.Distance(Directions[1].transform.position + Directions[1].transform.up * -selectionSizeAdjust, transform.position);
        //right
        array[2] = Vector3.Distance(Directions[2].transform.position + Directions[2].transform.right * selectionSizeAdjust, transform.position);
        //left
        array[3] = Vector3.Distance(Directions[2].transform.position + Directions[2].transform.right * -selectionSizeAdjust, transform.position);
        //if the controller is moved far enough from the saved position, the selection is evaluated, think of it like a deadzone
        if (Vector3.Distance(savedPosition, transform.position) > selectionThreshold)
        {
            //we proceed to go check for the smallest value in the array which equals the distance of the controller to the selection point
            float smallestValue = 99999; 
            for (int i = 0; i < array.Length; i++)
            {
                if(array[i] < smallestValue)
                {
                    smallestValue = array[i];
                    selection = i;
                }
            }
        }
        else
        {
            selection = -1;
        }
        yield return null;
    }      
}

// changes the material of the spell selection wheel to show the selection and reflect the element color
IEnumerator materialChanger()
{
    var intensity = 2f;
    Material[] materials = new Material[4];

    for (int i = 0; i < Directions.Length; i++)
    {
        materials[i] = Directions[i].GetComponent<ParticleSystemRenderer>().material;
    }
    
    foreach (var t in materials)
    {
        var color = t.GetColor("_TintColor");
        color = ofudaObjects[currentSpellSet]._Material.GetColor("_DissolveEmissionColor");
        color = new Color(color.r * intensity, color.g * intensity, color.b * intensity);
        t.SetColor("_TintColor",color);
    }
    
    while (isSelecting)
    {
        for (int i = 0; i < materials.Length; i++)
        {
            var color = materials[i].GetColor("_TintColor");
            if (i == selection)
            {
                color.a = 1;
            }
            else
            {
                color.a = 0.02f;
            }
            materials[i].SetColor("_TintColor",color);
        }
        yield return null;
    }
}

AI-Agent

Since I mentioned the AI-Agent let me quickly explain what it does. The AI-Agent is a script that is attached to every AI in the game, it handles the AI’s stats, the AI’s health and the AI’s target and targeting ranges like “IsInChaseRange”. It also handles the Damage logic when the AI gets hit and notifies the behaviour tree if it is dead. It also handles communication with the save system.

3rd Person Controller

The 3rd Person Controller is a bit of a separate thing from the rest of the game, it was only created as requirement for the project as requested by the teacher.
The controller is made up of two parts, the movement controller which lets you run around and jump.

public class PlayerController : MonoBehaviour
{
    [Header("Movement Variables")]
    
    //A variable that Controls move speed
    [Tooltip("Controls the players movement speed")]
    [SerializeField] private float moveSpeed = 5f;

    //the power/velocity of the jump
    [Tooltip("The Velocity/power of the jump")]
    [SerializeField] private float jumpPower = 10f;
    
    //Controls the amount of control left to the player while airborne, lower value is more control
    [Tooltip("the amount of control left to the player while airborne 5 - 15 recommended, a lower value is more control ")] 
    [SerializeField] private float airControl = 10;
    
    //variable that adjusts rotation speed
    [Tooltip("Controls the rotation speed")]
    [SerializeField] float rotationSpeed = 500f;

    [Header("Ground Check Settings")]
    
    //variable that decides the radius of the grounded check sphere
    [Tooltip("Decides the radius of the grounded check sphere")]
    [SerializeField] private float groundCheckRadius = 0.2f;
    
    //variable that controls the offset of the sphere to the player position
    [Tooltip("The offset of the ground check sphere")]
    [SerializeField] private Vector3 groundCheckOffset;

    //sets the layers the player checks for ground collision
    [Tooltip("The layers that are being checked for collision")]
    [SerializeField] private LayerMask groundLayer;

    //Camera controller grabbed from the main camera in the scene
    private CameraController cameraController;
    
    //Animator grabbed from the current player character
    private Animator animator;
    
    //a variable checking if the player is using any movement input NOT if the player has any remaining velocity
    private bool isMoving = false;

    //the amount of movement being applied clamped to 0 - 1
    private float moveAmount;
    
    //a variable to adjust the players target rotation controlled by the cameras view direction to allow to smoothly turn towards it
    private Quaternion targetRotation;
    
    //is true if the player is touching the ground or the layer set as ground with its feet
    public bool isGrounded;
    
    //is true if the player starts being airborne through the jump action
    private bool isJumping;

    //keeps track of where in the combo we are
    private int comboCounter = 0;
    
    //the vertical velocity of the player 
    float yVelocity = 0f;
    
    //saved momentum, when airborne this keeps launching you in the direction you had momentum towards
    private Vector3 savedMotion;
    
    //the character controller on the player
    private CharacterController m_CharacterController;

    //needed to check if we are attacking
    private PlayerCombat m_PlayerCombat;
    
    //checks if we should be allowed to move and jump
    private bool isBusy;
    
    //if the player has turned after attacking
    private bool hasTurned;
    

    private void Awake()
    {
        //Find references to components
        cameraController = Camera.main.GetComponent<CameraController>();
        animator = GetComponent<Animator>();
        m_CharacterController = GetComponent<CharacterController>();
        m_PlayerCombat = GetComponent<PlayerCombat>();
    }

    private void Update()
    {
        //Checks if the player is grounded
        GroundCheck();
        if (m_PlayerCombat.isAttacking)
        {
            isBusy = true;
        }
        else
        {
            isBusy = false;
        }
        //vertical and horizontal input gets turned into move direction
        var moveDir = Input();
        
        
        //calculates and applies movement
        Move(moveDir);
        
        //handles the animation states
        AnimationHandler();

    }

    private Vector3 Input()
    {
        //Input via mouse or controller set in the unity input system
        float h = UnityEngine.Input.GetAxis("Horizontal");
        float v = UnityEngine.Input.GetAxis("Vertical");
        
        //jump input
        if (UnityEngine.Input.GetButtonDown("Jump"))
        {
            isJumping = true;
        }
        else
        {
            if (isGrounded)
            {
                isJumping = false;
            }
        }
        

        //calculates the forward direction the player should be moving too
        var moveInput = (new Vector3(h, 0, v)).normalized;
        
        //calculates the amount of movement
        CalculatesMoveAmount(h, v);
        
        //multiplies the move input with the camera view direction on the planar X,Z axis to create a move direction
        var moveDir = cameraController.PlanarRotation * moveInput;
        if (m_PlayerCombat.isAttacking && moveDir == Vector3.zero) { moveDir = cameraController.PlanarRotation * Vector3.forward;}
        return moveDir;
    }

    
    // handles both the moveAmount and isMoving variable
    private void CalculatesMoveAmount(float h, float v)
    {
        //clamps the move amount to 0 - 1
        moveAmount = Mathf.Clamp01(Mathf.Abs(h) + Mathf.Abs(v));

        //checks if we are moving
        if (moveAmount > 0 && !isBusy)
        {
            isMoving = true;
        }
        else {
            isMoving = false;
        }
    }
    
    //sets Parameters in the animator
    private void AnimationHandler()
    {
        //set moveAmount variable to the animator
        animator.SetFloat("moveAmount",moveAmount,0.2f,Time.deltaTime);

        if (isGrounded)
        {
            //grounded
            animator.SetBool("isGrounded", true);
            
            //falling
            animator.SetBool("isFalling", false);
            
        }
        else
        {
            // not grounded
            animator.SetBool("isGrounded", false);
            
            //sets falling to true if we are jumping or falling a certain speed
            if (isJumping || yVelocity < -3.5f)
            {
                animator.SetBool("isFalling", true);
            }
        }
        
        //if we are jumping or not
        animator.SetBool("isJumping", isJumping);
    }
    
    

    private void Move(Vector3 moveDir)
    {
        //apply moveSpeed to the move direction to create a movement velocity vector
        var velocity = moveDir * moveSpeed;
        if (isBusy) { velocity = Vector3.zero;}
        if (isGrounded)
        {
            //save the motion of our current movement velocity in case we leave the ground and need to keep using this velocity
            savedMotion = velocity;
            
            //if we are grounded we are not falling
            yVelocity = 0f;
            
            //if the player is jumping the jump power is being applied
            if (isJumping && !isBusy)
            {
                yVelocity = jumpPower;
            }
        }
        else
        {
            //apply gravity to the players Y/Vertical Velocity
            yVelocity += Physics.gravity.y * Time.deltaTime;
        }

        if (!isGrounded)
        {
            //momentum continuation when falling/jumping and the control you keep over your character
            velocity = Vector3.ClampMagnitude(velocity/(airControl * 100) + savedMotion, moveSpeed);
            
            //the new velocity is saved as our new saved momentum to make sure we can actually make an impact to the initial momentum
            savedMotion = velocity;
        }
        // the calculated Y/Vertical Velocity actually gets applied to the main Velocity
        velocity.y = yVelocity;
        
        //the character controller move function where we apply our calculated velocity to the actual transform
        m_CharacterController.Move(velocity * Time.deltaTime);
        
        //if we are moving the character or attacking should look at where the camera is pointing here the target for said rotation is being calculated
        RotatePlayer(moveDir);
    }

    private void RotatePlayer(Vector3 moveDir)
    {
        if (isMoving || m_PlayerCombat.isAttacking && !hasTurned)
        {
            targetRotation = Quaternion.LookRotation(moveDir);
        }
        if (m_PlayerCombat.isAttacking && !hasTurned)
        {
            hasTurned = true;
        }

        if (hasTurned && !m_PlayerCombat.isAttacking)
        {
            hasTurned = false;
        }

        //and here said target rotation is being applied to the actual transform over time for a smooth rotation
        transform.rotation = Quaternion.RotateTowards(transform.rotation, targetRotation, rotationSpeed * Time.deltaTime);
    }

    //checks if the player is colliding with the ground
    void GroundCheck()
    {
        //checks if the sphere is colliding with any ground layer object
        isGrounded = Physics.CheckSphere(transform.TransformPoint(groundCheckOffset), groundCheckRadius, groundLayer);
    }

    //draws a gizmo visualization for the groundCheck Sphere
    private void OnDrawGizmosSelected()
    {
        Gizmos.color = new Color(0, 1, 0, 0.5f);
        Gizmos.DrawSphere(transform.TransformPoint(groundCheckOffset), groundCheckRadius);
    }
}

public class PlayerCombat : MonoBehaviour
{
    [SerializeField] List<AttackSO> combo;

    [Header("Animation Settings")]
    [SerializeField] [Range(0f,1f)] private float _AnimationPlayCheck = .5f;
    [SerializeField] [Range(0f,1f)] private float _AnimationAmountPlayedCheck = .95f;
    [SerializeField] private float _AnimationTransition = 0.25f;
    
    [Header("Combo Settings")]
    [SerializeField] private float _ComboStartDelay = 0.4f;
    [SerializeField] private float _ComboAttackDelay = 0f;
    [SerializeField] private float _ComboEndDelay = 0.2f;   
  
    [HideInInspector]public float damage;
    [HideInInspector] public bool isAttacking;
    private float lastComboBegin;
    private float lastComboEnd;
    private int comboCounter;
    private Animator _animator;
    private PlayerController _playerController;
    private bool hasAttacked = false;
    private bool flip;
    
    protected AnimatorOverrideController animatorOverrideController;
    protected AnimationClipOverrides clipOverrides;
    
    // Start is called before the first frame update
    private void Awake()
    {
        _animator = GetComponent<Animator>();
        _playerController = GetComponent<PlayerController>();
    }

    private void Start()
    {
        animatorOverrideController = new AnimatorOverrideController(_animator.runtimeAnimatorController);
        _animator.runtimeAnimatorController = animatorOverrideController;

        clipOverrides = new AnimationClipOverrides(animatorOverrideController.overridesCount);
        animatorOverrideController.GetOverrides(clipOverrides);
    }

    // Update is called once per frame
    void Update()
    {
        ExitAttack();
        isAttacking = comboCounter > 0;
        if (!_playerController.isGrounded || _animator.IsInTransition(0)) return;
        if (Input.GetButton("Fire1"))
        {
            Attack();
        }

        if (!hasAttacked && comboCounter > 0 && _animator.GetCurrentAnimatorStateInfo(0).normalizedTime >= 0.75)
        {
            ExecuteAttack();
        }
    }

    void Attack()
    {
        if (Time.time - lastComboEnd > _ComboStartDelay && comboCounter < combo.Count)
        {
            CancelInvoke("EndCombo");

            if (Time.time - lastComboBegin >= _ComboAttackDelay)
            {
                if (_animator.GetCurrentAnimatorStateInfo(0).normalizedTime > _AnimationPlayCheck && comboCounter > 0 && hasAttacked)
                {
                    hasAttacked = false;
                }
                else
                {
                    if (comboCounter == 0)
                    {
                        ExecuteAttack();
                    }
                }
            }
        }
    }
    private void ExecuteAttack()
    {
        hasAttacked = true;
        CancelInvoke("EndCombo");

        _animator.SetFloat("attackSpeed",combo[comboCounter].animationSpeed);
        _animator.SetBool("isMirrored",combo[comboCounter].isMirrored);

        switch (flipSwitch())
        {
            case true:
                clipOverrides["Attack1"] = combo[comboCounter].AttackAnimationClip;
                animatorOverrideController.ApplyOverrides(clipOverrides);
                _animator.CrossFadeInFixedTime("Attack1",_AnimationTransition);
                
                
                break;
            case false:
                clipOverrides["Attack2"] = combo[comboCounter].AttackAnimationClip;
                _animator.CrossFadeInFixedTime("Attack2",_AnimationTransition);
                animatorOverrideController.ApplyOverrides(clipOverrides);
                
                break;

        }
        
        //Change damage depending on attack (has yet to be implemented)
        damage = combo[comboCounter].damage;
        lastComboBegin = Time.time;
        comboCounter++;
    }

    private void UpdateOverrides()
    {
    
    }

    void ExitAttack()
    {
        if (!_animator.IsInTransition(0))
        {
            if (_animator.GetCurrentAnimatorStateInfo(0).normalizedTime >= 0.99 && _animator.GetCurrentAnimatorStateInfo(0).IsTag("Attack"))
            {
                Invoke("EndCombo",_ComboEndDelay);
            }
            
        }
        
        
    }

    void EndCombo()
    {
        comboCounter = 0;
        flip = false;
        lastComboEnd = Time.time;
    }

    bool flipSwitch()
    {
        flip = !flip;
        return flip;
    }  
}

Water Shader

The water shader I made using amplify, is an advanced shader using math to calculate accurate waves, in addition with refraction and depth to simulate water.

The node setup looks as following:
The Custom node GerstnerWave being the following code: 
float steepness = wave.z;
float wavelength = wave.w;
float k = 2 * 3.141593 / wavelength;
float c = sqrt(9.8 / k);
float2 d = normalize(wave.xy);
float f = k * (dot(d, p.xz) - c * _Time.y);
float a = steepness / k;

tangent += float3(
    -d.x * d.x * (steepness * sin(f)),
    d.x * (steepness * cos(f)),
    -d.x * d.y * (steepness * sin(f))
);
binormal += float3(
    -d.x * d.y * (steepness * sin(f)),
    d.y * (steepness * cos(f)),
    -d.y * d.y * (steepness * sin(f))
);
return float3(
    d.x * (a * cos(f)),
    a * sin(f),
    d.y * (a * cos(f))
);
